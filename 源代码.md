358      private void loadImageUrls(String imageUrlsJson, List<String> targetList) {
359          targetList.clear();
360          if (imageUrlsJson == null || imageUrlsJson.isEmpty()) {
361              return;
362          }
363          
364          try {
365              // 尝试解析JSON数组
366              Gson gson = new Gson();
367              List<String> urls = gson.fromJson(imageUrlsJson, new TypeToken<List<String>>(){}.getType());
368              if (urls != null) {
369                  targetList.addAll(urls);
370              }
371          } catch (Exception e) {
372              // 如果不是JSON，尝试按逗号分隔
373              String[] urls = imageUrlsJson.split(",");
374              for (String url : urls) {
375                  String trimmed = url.trim();
376                  if (!trimmed.isEmpty()) {
377                      targetList.add(trimmed);
378                  }
379              }
380          }
381      }

74           tvTitle = findViewById(R.id.tvTitle);

76           btnMarkAllRead = findViewById(R.id.btnMarkAllRead);
77           tvEmptyState = findViewById(R.id.tvEmptyState);
78       }
79   
80       private void setupRecyclerView() {
81           adapter = new NotificationAdapter(notificationList);
82           rvNotifications.setLayoutManager(new LinearLayoutManager(this));
83           rvNotifications.setAdapter(adapter);
84       }
85   
86       private void loadNotificationData() {
87           notificationApi.getAllNotifications().enqueue(new Callback<ApiResponse<List<Notification>>>() {
88               @Override
89               public void onResponse(Call<ApiResponse<List<Notification>>> call, Response<ApiResponse<List<Notification>>> response) {
90                   runOnUiThread(() -> {
91                       notificationList.clear();
92                       if (response.isSuccessful() && response.body() != null && response.body().getCode() == 200) {
93                           notificationList.addAll(response.body().getData());
94                       } else {
95                           Toast.makeText(MessageNotificationActivity.this, "获取通知数据失败: " + (response.body()!=null?response.body().getMessage():"接口异常"), Toast.LENGTH_LONG).show();
96                       }
97                       adapter.notifyDataSetChanged();
98                       updateEmptyState();
99                   });
100              }
101              @Override
102              public void onFailure(Call<ApiResponse<List<Notification>>> call, Throwable t) {
103                  runOnUiThread(() -> {
104                      Toast.makeText(MessageNotificationActivity.this, "获取通知数据失败: 网络异常", Toast.LENGTH_LONG).show();
105                      notificationList.clear();
106                      adapter.notifyDataSetChanged();
107                      updateEmptyState();
108                  });
109              }
110          });
111      }
112  
113      private void updateEmptyState() {
114          if (notificationList.isEmpty()) {
115              tvEmptyState.setVisibility(View.VISIBLE);
116              rvNotifications.setVisibility(View.GONE);
117              btnMarkAllRead.setVisibility(View.GONE);
118          } else {
119              tvEmptyState.setVisibility(View.GONE);
120              rvNotifications.setVisibility(View.VISIBLE);
121              btnMarkAllRead.setVisibility(View.VISIBLE);
122          }
123      }

143          // 示例：调用notificationApi.markAllAsRead();
