void select(int arr[],int n)
{
    for(int i=0;i<n;i++)
    {
        int min=i;
        int j;
        for(j=i;j<n;j++)
        {
            if(arr[j]<arr[min])
            {
                min=j;
            }
        }
        swap(arr[i],arr[min]);
    }
}

# 2.冒泡排序

![img](https://i-blog.csdnimg.cn/blog_migrate/bfe2e655008710a852274f86240c6963.gif)

说明：

1.每次比较两个相邻的数字的大小，把大的数往后移

2.比较n-1次即可

代码:

```c++
void bubble(int arr[], int n)
{
    for(int i=0;i<n-1;i++)//排序次数，n个数只需比价n-1次
    {
        for(int j=0;j<n-1-i;j++)//每次中依次比较两个相邻元素,每次找到最大的数放在最后
        {
            if(arr[j]>arr[j+1])
            {
                int temp=arr[j];
                arr[j]=arr[j+1];
                arr[j+1]=temp;
            }
        }
    }
}
```

# 3.归并排序

<img src="https://i-blog.csdnimg.cn/blog_migrate/17847222181095acdab5c4a31b307b16.gif" alt="img" style="zoom:50%;" />

说明：

1.先不断的的分割数据，形成一个个子序列

2.当所有的子序列有序的时候，不断的合并子序列，形成更大的新的子序列，最终让整个数组有序

代码：

```c++
//合并两个有序数组,左子数组和右子数组都是有序数组
void merge(int arr[],int left,int mid,int right){
    int i,j,k;
    //计算左右两个子数组的长度
    int n1=mid-left+1;
    int n2=right-mid;
    //创建两个临时数组L和R用来后续比较左右子数组的元素
    int L[n1],R[n2];
    for(i=0;i<n1;i++)
    {
        L[i]=arr[left+i];
    }
    for(j=0;j<n2;j++)
    {
        R[j]=arr[mid+1+j];
    }
    i=0;j=0;k=left;
    while(i<n1&&j<n2)
    {
        if(L[i]<R[j]){
            arr[k++]=L[i++];
        }
        else{
            arr[k++]=R[j++];
        }
        
    }
    //将剩余的元素复制到arr中
    while(i<n1)
    {
        arr[k++]=L[i++];
    }
    while(j<n2)
    {
        arr[k++]=R[j++];
    }

}
void mergesort(int arr[],int left,int right){
    if(left<right)
    {
        int mid=(left+right)/2;
        //递归调用，分别对左右子数组进行排序
        mergesort(arr,left,mid);
        mergesort(arr,mid+1,right);
        //合并两个有序数组
        merge(arr,left,mid,right);
    }
}

```



# 4.插入排序

<img src="https://i-blog.csdnimg.cn/blog_migrate/7b23514289fd4c0fa3eeb9343a58baad.gif" alt="img" style="zoom:50%;" />

说明：

1.数组组成： 已排好序列+未排序列

2.先判断i位置的数是否比前一个数小，如果当前数比前一位数字还大，无需比较

3.如果当前数字比前一位数字小，**已排好序列中比这个数大的自动向后移动一位**，插入到已排好的序列中形成新的已排好序列

代码：

```c++
void insert(int arr[],int n)
{
    for(int i=1;i<n;i++)//外部循环完成插入排序
    {
        if(arr[i]<arr[i-1])//先判断，如果i位置的数比前一个数小，那么才需要插入排序，否则不需要
        {
            int temp=arr[i];
            int j;
            for(j=i-1;j>=0&&arr[j]>temp;j--)//内部循环完成一次的插入
            {
                arr[j+1]=arr[j];//将比temp大的数往后移动一位
            }
            arr[j+1]=temp;//内部循环完成一次的插入
        }
    }
}
//两次循环实现
//内部循环完成一次的插入
//外层循环完成插入排序
```

# 6.希尔排序

![img](https://i-blog.csdnimg.cn/blog_migrate/93fae629acfc03414c2645e4c160ff01.gif)

说明：
1.插入排序的优化版，先进行预排序，让大的数快速跳到后面，小的数快速跳到前面，、

2.然后对该数列进行一次插入排序

3.相当于把直接插入排序中的1换成gap

```c++
void shell(int arr[],int n)
{
    int gap;
    for(gap=n/2;gap>0;gap/=2)//初始间隔为数组的一半，每次将间隔减半，直到间隔为0
    {
        for(int i=gap;i<n;i++)//对每个间隔进行插入排序
        {
            int temp=arr[i];
            int j;
            for(j=i;j>=gap&&arr[j-gap]>temp;j-=gap)
            {
                arr[j]=arr[j-gap];
            }
            arr[j]=temp;
        }
    }
}
```
